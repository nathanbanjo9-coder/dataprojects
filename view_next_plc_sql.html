-- STORES TABLE
CREATE TABLE stores (
    store_id VARCHAR(10) PRIMARY KEY,
    store_name VARCHAR(100),
    store_type VARCHAR(50),
    region VARCHAR(50),
    city VARCHAR(50),
    postcode VARCHAR(10),
    opened_year INT,
    sq_footage INT
);

-- PRODUCTS TABLE
CREATE TABLE products (
    product_id VARCHAR(10) PRIMARY KEY,
    product_name VARCHAR(150),
    category VARCHAR(50),
    subcategory VARCHAR(50),
    cost_price NUMERIC(10,2),
    sell_price NUMERIC(10,2),
    margin_pct NUMERIC(5,2),
    is_online_only BOOLEAN,
    season VARCHAR(20)
);

-- CUSTOMERS TABLE
CREATE TABLE customers (
    customer_id VARCHAR(10) PRIMARY KEY,
    gender VARCHAR(20),
    age_band VARCHAR(10),
    region VARCHAR(50),
    next_pay_holder BOOLEAN,
    loyalty_member BOOLEAN,
    acquisition_year INT
);

-- ORDERS TABLE
CREATE TABLE orders (
    order_id VARCHAR(15) PRIMARY KEY,
    order_date DATE,
    customer_id VARCHAR(10),
    product_id VARCHAR(10),
    category VARCHAR(50),
    subcategory VARCHAR(50),
    channel VARCHAR(20),
    store_id VARCHAR(10),
    region VARCHAR(50),
    quantity INT,
    sell_price NUMERIC(10,2),
    discount_pct INT,
    revenue NUMERIC(10,2),
    cost NUMERIC(10,2),
    gross_profit NUMERIC(10,2),
    is_returned BOOLEAN,
    return_reason VARCHAR(50),
    next_pay_used BOOLEAN,
    year INT,
    month INT,
    quarter VARCHAR(5),
    day_of_week VARCHAR(10)
);

-- RETURNS TABLE
CREATE TABLE returns (
    return_id VARCHAR(10) PRIMARY KEY,
    order_id VARCHAR(15),
    order_date DATE,
    customer_id VARCHAR(10),
    product_id VARCHAR(10),
    category VARCHAR(50),
    channel VARCHAR(20),
    region VARCHAR(50),
    revenue NUMERIC(10,2),
    return_reason VARCHAR(50),
    return_date DATE
);

-- ── SECTION 1: DATA VALIDATION ───────────────────────────────

-- Q1: Check for duplicate order IDs
SELECT order_id, COUNT(*) AS duplicate_count
FROM orders
GROUP BY order_id
HAVING COUNT(*) > 1;

-- Q2: Check for nulls across key columns
SELECT
    COUNT(*) FILTER (WHERE order_id IS NULL)     AS null_order_ids,
    COUNT(*) FILTER (WHERE order_date IS NULL)   AS null_dates,
    COUNT(*) FILTER (WHERE customer_id IS NULL)  AS null_customers,
    COUNT(*) FILTER (WHERE revenue IS NULL)      AS null_revenue,
    COUNT(*) FILTER (WHERE category IS NULL)     AS null_category
FROM orders;

-- Q3: Validate revenue logic (sell price x quantity should equal revenue)
SELECT
    order_id,
    sell_price,
    quantity,
    revenue,
    ROUND((sell_price * quantity)::NUMERIC, 2) AS expected_revenue,
    ROUND((revenue - (sell_price * quantity))::NUMERIC, 2) AS variance
FROM orders
WHERE ROUND((revenue - (sell_price * quantity))::NUMERIC, 2) <> 0
LIMIT 20;

-- Q4: Check date range integrity
SELECT
    MIN(order_date) AS earliest_order,
    MAX(order_date) AS latest_order,
    COUNT(DISTINCT order_date) AS distinct_days
FROM orders;


-- ── SECTION 2: COMMERCIAL KPIS ───────────────────────────────

-- Q5: Headline KPIs — total revenue, profit, orders, AOV
SELECT
    COUNT(order_id)                          AS total_orders,
    ROUND(SUM(revenue)::NUMERIC, 2)          AS total_revenue,
    ROUND(SUM(gross_profit)::NUMERIC, 2)     AS total_gross_profit,
    ROUND(AVG(revenue)::NUMERIC, 2)          AS avg_order_value,
    ROUND(SUM(gross_profit) / NULLIF(SUM(revenue), 0) * 100, 2) AS gross_margin_pct
FROM orders;

-- Q6: Revenue and profit by category
SELECT
    category,
    COUNT(order_id)                          AS total_orders,
    ROUND(SUM(revenue)::NUMERIC, 2)          AS total_revenue,
    ROUND(SUM(gross_profit)::NUMERIC, 2)     AS gross_profit,
    ROUND(SUM(gross_profit) / NULLIF(SUM(revenue), 0) * 100, 2) AS margin_pct
FROM orders
GROUP BY category
ORDER BY total_revenue DESC;

-- Q7: Revenue by channel
SELECT
    channel,
    COUNT(order_id)                          AS total_orders,
    ROUND(SUM(revenue)::NUMERIC, 2)          AS total_revenue,
    ROUND(SUM(revenue) / SUM(SUM(revenue)) OVER () * 100, 2) AS revenue_share_pct
FROM orders
GROUP BY channel
ORDER BY total_revenue DESC;

-- Q8: Monthly revenue trend with month on month growth
WITH monthly AS (
    SELECT
        year,
        month,
        ROUND(SUM(revenue)::NUMERIC, 2) AS monthly_revenue
    FROM orders
    GROUP BY year, month
)
SELECT
    year,
    month,
    monthly_revenue,
    LAG(monthly_revenue) OVER (ORDER BY year, month) AS prev_month_revenue,
    ROUND(
        (monthly_revenue - LAG(monthly_revenue) OVER (ORDER BY year, month))
        / NULLIF(LAG(monthly_revenue) OVER (ORDER BY year, month), 0) * 100
    , 2) AS mom_growth_pct
FROM monthly
ORDER BY year, month;


-- ── SECTION 3: RETURNS ANALYSIS ──────────────────────────────

-- Q9: Overall returns rate
SELECT
    COUNT(order_id)                                              AS total_orders,
    SUM(CASE WHEN is_returned THEN 1 ELSE 0 END)                AS total_returns,
    ROUND(SUM(CASE WHEN is_returned THEN 1 ELSE 0 END)::NUMERIC
        / COUNT(order_id) * 100, 2)                             AS return_rate_pct,
    ROUND(SUM(CASE WHEN is_returned THEN revenue ELSE 0 END)::NUMERIC, 2) AS revenue_at_risk
FROM orders;

-- Q10: Returns rate by category and channel
SELECT
    category,
    channel,
    COUNT(order_id)                                              AS total_orders,
    SUM(CASE WHEN is_returned THEN 1 ELSE 0 END)                AS returns,
    ROUND(SUM(CASE WHEN is_returned THEN 1 ELSE 0 END)::NUMERIC
        / COUNT(order_id) * 100, 2)                             AS return_rate_pct
FROM orders
GROUP BY category, channel
ORDER BY return_rate_pct DESC;

-- Q11: Top return reasons with revenue impact
SELECT
    return_reason,
    COUNT(return_id)                             AS total_returns,
    ROUND(SUM(revenue)::NUMERIC, 2)              AS revenue_lost,
    ROUND(SUM(revenue) / SUM(SUM(revenue)) OVER () * 100, 2) AS pct_of_total_returns
FROM returns
GROUP BY return_reason
ORDER BY revenue_lost DESC;


-- ── SECTION 4: REGIONAL AND STORE PERFORMANCE ────────────────

-- Q12: Revenue and margin by region
SELECT
    o.region,
    COUNT(o.order_id)                            AS total_orders,
    ROUND(SUM(o.revenue)::NUMERIC, 2)            AS total_revenue,
    ROUND(SUM(o.gross_profit)::NUMERIC, 2)       AS gross_profit,
    ROUND(SUM(o.gross_profit) / NULLIF(SUM(o.revenue), 0) * 100, 2) AS margin_pct
FROM orders o
GROUP BY o.region
ORDER BY total_revenue DESC;

-- Q13: Top 10 performing stores by revenue
SELECT
    s.store_name,
    s.region,
    s.store_type,
    COUNT(o.order_id)                            AS total_orders,
    ROUND(SUM(o.revenue)::NUMERIC, 2)            AS total_revenue,
    ROUND(SUM(o.gross_profit)::NUMERIC, 2)       AS gross_profit
FROM orders o
JOIN stores s ON o.store_id = s.store_id
GROUP BY s.store_name, s.region, s.store_type
ORDER BY total_revenue DESC
LIMIT 10;

-- Q14: Region performance ranked using window function
WITH region_revenue AS (
    SELECT
        region,
        ROUND(SUM(revenue)::NUMERIC, 2) AS total_revenue,
        ROUND(SUM(gross_profit)::NUMERIC, 2) AS gross_profit
    FROM orders
    GROUP BY region
)
SELECT
    region,
    total_revenue,
    gross_profit,
    RANK() OVER (ORDER BY total_revenue DESC)   AS revenue_rank,
    RANK() OVER (ORDER BY gross_profit DESC)    AS profit_rank
FROM region_revenue;


-- ── SECTION 5: PRODUCT ANALYTICS ─────────────────────────────

-- Q15: Top 20 products by revenue with margin
SELECT
    p.product_name,
    p.category,
    p.subcategory,
    COUNT(o.order_id)                            AS times_ordered,
    ROUND(SUM(o.revenue)::NUMERIC, 2)            AS total_revenue,
    ROUND(SUM(o.gross_profit)::NUMERIC, 2)       AS gross_profit,
    ROUND(AVG(o.discount_pct)::NUMERIC, 2)       AS avg_discount_pct
FROM orders o
JOIN products p ON o.product_id = p.product_id
GROUP BY p.product_name, p.category, p.subcategory
ORDER BY total_revenue DESC
LIMIT 20;

-- Q16: Discount impact on margin by category
SELECT
    category,
    discount_pct,
    COUNT(order_id)                              AS total_orders,
    ROUND(AVG(gross_profit)::NUMERIC, 2)         AS avg_gross_profit,
    ROUND(SUM(gross_profit) / NULLIF(SUM(revenue), 0) * 100, 2) AS margin_pct
FROM orders
GROUP BY category, discount_pct
ORDER BY category, discount_pct;


-- ── SECTION 6: CUSTOMER ANALYTICS ────────────────────────────

-- Q17: Revenue by customer age band and gender
SELECT
    c.age_band,
    c.gender,
    COUNT(DISTINCT o.customer_id)                AS unique_customers,
    COUNT(o.order_id)                            AS total_orders,
    ROUND(SUM(o.revenue)::NUMERIC, 2)            AS total_revenue,
    ROUND(AVG(o.revenue)::NUMERIC, 2)            AS avg_order_value
FROM orders o
JOIN customers c ON o.customer_id = c.customer_id
GROUP BY c.age_band, c.gender
ORDER BY total_revenue DESC;

-- Q18: NextPay vs non NextPay customer spend comparison
SELECT
    c.next_pay_holder,
    COUNT(DISTINCT o.customer_id)                AS unique_customers,
    COUNT(o.order_id)                            AS total_orders,
    ROUND(SUM(o.revenue)::NUMERIC, 2)            AS total_revenue,
    ROUND(AVG(o.revenue)::NUMERIC, 2)            AS avg_order_value,
    ROUND(AVG(o.discount_pct)::NUMERIC, 2)       AS avg_discount_pct
FROM orders o
JOIN customers c ON o.customer_id = c.customer_id
GROUP BY c.next_pay_holder
ORDER BY total_revenue DESC;


-- ── SECTION 7: ADVANCED WINDOW FUNCTIONS & CTES ──────────────

-- Q19: Cumulative revenue by month using running total
SELECT
    year,
    month,
    ROUND(SUM(revenue)::NUMERIC, 2) AS monthly_revenue,
    ROUND(SUM(SUM(revenue)) OVER (
        PARTITION BY year ORDER BY month
    )::NUMERIC, 2) AS cumulative_revenue_ytd
FROM orders
GROUP BY year, month
ORDER BY year, month;

-- Q20: Year on year revenue comparison by category
WITH yearly AS (
    SELECT
        category,
        year,
        ROUND(SUM(revenue)::NUMERIC, 2) AS annual_revenue
    FROM orders
    GROUP BY category, year
)
SELECT
    category,
    year,
    annual_revenue,
    LAG(annual_revenue) OVER (PARTITION BY category ORDER BY year) AS prev_year_revenue,
    ROUND(
        (annual_revenue - LAG(annual_revenue) OVER (PARTITION BY category ORDER BY year))
        / NULLIF(LAG(annual_revenue) OVER (PARTITION BY category ORDER BY year), 0) * 100
    , 2) AS yoy_growth_pct
FROM yearly
ORDER BY category, year;

-- Q21: Identify high value customers using RANK
WITH customer_spend AS (
    SELECT
        customer_id,
        COUNT(order_id)                          AS total_orders,
        ROUND(SUM(revenue)::NUMERIC, 2)          AS total_spend,
        ROUND(AVG(revenue)::NUMERIC, 2)          AS avg_order_value,
        SUM(CASE WHEN is_returned THEN 1 ELSE 0 END) AS total_returns
    FROM orders
    GROUP BY customer_id
)
SELECT
    customer_id,
    total_orders,
    total_spend,
    avg_order_value,
    total_returns,
    RANK() OVER (ORDER BY total_spend DESC) AS spend_rank
FROM customer_spend
ORDER BY spend_rank
LIMIT 50;

-- Q22: Seasonal performance — revenue by quarter and category
SELECT
    quarter,
    category,
    ROUND(SUM(revenue)::NUMERIC, 2)              AS total_revenue,
    ROUND(SUM(gross_profit)::NUMERIC, 2)         AS gross_profit,
    RANK() OVER (
        PARTITION BY quarter ORDER BY SUM(revenue) DESC
    ) AS category_rank_within_quarter
FROM orders
GROUP BY quarter, category
ORDER BY quarter, category_rank_within_quarter;

-- Check headline KPIs came back
SELECT COUNT(order_id) AS total_orders,
ROUND(SUM(revenue)::NUMERIC, 2) AS total_revenue
FROM orders;

-- Check revenue by category
SELECT category, ROUND(SUM(revenue)::NUMERIC, 2) AS total_revenue
FROM orders
GROUP BY category
ORDER BY total_revenue DESC;

-- Check returns rate
SELECT ROUND(SUM(CASE WHEN is_returned THEN 1 ELSE 0 END)::NUMERIC
/ COUNT(order_id) * 100, 2) AS return_rate_pct
FROM orders;
